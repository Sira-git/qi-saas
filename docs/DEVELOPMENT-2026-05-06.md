```javascript
// 批量操作
window.confirm(`确定要${actionText}选中的 ${count} 个企业吗？`)

window.confirm('确定要重置该企业账号的密码吗？新密码将设置为 123456')

// 重新生成报告
window.confirm('确定要重新生成该报告吗？')
```

### 加载状态
```javascript
loading.value = true  // 开始加载
try {
  // 执行操作
  await new Promise(resolve => setTimeout(resolve, 800))
  message.success('操作成功')
} catch (e) {
  message.error('操作失败')
} finally {
  loading.value = false
}
```

### 表单验证
```javascript
if (!form.name) {
  message.warning('请输入企业名称')
  return
}
```
